palette.pages = pages;
palette.open = false;
palette.currentPage = 'root';

const toggleBtn = document.getElementById('toggle-btn');
toggleBtn.addEventListener('click', () => {
  palette.open = !palette.open;
});

palette.addEventListener('cmdk-select', (e) => {
  console.log('Selected:', e.detail);
  palette.open = false;
});

palette.addEventListener('cmdk-page', (e) => {
  console.log('Page changed:', e.detail);
});

document.addEventListener('keydown', (e) => {
  if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
    e.preventDefault();
    palette.open = !palette.open;
  }
});
import { useState } from 'react';
import type { CmdkPageData } from 'cmdk-wc';

const pages: CmdkPageData[] = [

    id: 'root',

      {
        id: 'main',
        heading: 'Main',
        items: [
          { id: 'home', label: 'Home', icon: '🏠', href: '#' },
          { id: 'settings', label: 'Settings', icon: '⚙️', href: '#' },
          { id: 'projects', label: 'Projects', icon: '📁', page: 'projects', closeOnSelect: false },
        ],
      },
      {
        id: 'other',
        heading: 'Other',

      },
    ],
  },
];

export default function App() {
  const [open, setOpen] = useState(false);
  const [currentPage, setCurrentPage] = useState('root');

  const handleSelect = (e: any) => {
    console.log('Selected:', e.detail);
  };

  return (
    <div>
      <button onClick={() => setOpen(!open)}>
        {open ? 'Close Palette' : 'Open Palette'}
      </button>
      <cmdk-palette
        open={open}
        pages={pages}
        currentPage={currentPage}
        onCmdkSelect={handleSelect}
        onCmdkPage={(e: any) => setCurrentPage(e.detail.page)}
      />
    </div>
  ); }
<template>
  <div>
    <button @click="togglePalette">

    <cmdk-palette

      :pages="pages"

      @cmdkPage="handlePageChange"
      @open="open = true"
